117def feature_search(request): 

118 """ 

119 Search for a feature by title or description 

120 """ 

121 form = SearchForm 

122 query = None 

123 results = [] 

124 if 'query' in request.GET: 

125 form = SearchForm(request.GET) 

126 if form.is_valid(): 

127 query = form.cleaned_data['query'] 

128 results = Feature.objects.annotate(search=SearchVector('title', 'description')).filter(search=query) 

129 

130 return render(request, 'features/search.html', 

131 {'form': form, 

132 'query': query,
